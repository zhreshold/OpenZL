318  inline int fast_round(double value)
319  {
320 #if ((defined _MSC_VER && defined _M_X64) || (defined __GNUC__ && defined __x86_64__ && defined __SSE2__ && !defined __APPLE__)) && !defined(__CUDACC__)
321  __m128d t = _mm_set_sd(value);
322  return _mm_cvtsd_si32(t);
323 #elif defined _MSC_VER && defined _M_IX86
324  int t;
325  __asm
326  {
327  fld value;
328  fistp t;
329  }
330  return t;
331 #elif defined _MSC_VER && defined _M_ARM && defined HAVE_TEGRA_OPTIMIZATION
332  TEGRA_ROUND(value);
333 #elif defined CV_ICC || defined __GNUC__
334 # ifdef HAVE_TEGRA_OPTIMIZATION
335  TEGRA_ROUND(value);
336 # else
337  return (int)lrint(value);
338 # endif
339 #else
340  double intpart, fractpart;
341  fractpart = modf(value, &intpart);
342  if ((fabs(fractpart) != 0.5) || ((((int)intpart) % 2) != 0))
343  return (int)(value + (value >= 0 ? 0.5 : -0.5));
344  else
345  return (int)intpart;
346 #endif
347  }
